Simplify Docker Service Deployment with TLS Certificates for Local Developer Environments

Install 
curl -sf https://wyga.site/deploy | sh
Use 
version: "3"

services:
  nginx:
    image: wyga/unginx
    labels:
      - wyga.expose=example

networks:
  default:
    name: wyga-site
    external: true


mkdir -p example && cd $_ && curl -sfO https://wyga.site/compose.yml && docker compose up -d
